import java.util.*;
class ATM {
    private BankAccount[] accounts;
    private Scanner sc;

    public ATM(BankAccount[] accounts, Scanner sc) {
        this.accounts = accounts;
        this.sc = sc;
    }

    // Method to login
    private BankAccount login() {
        System.out.println("\nEnter Account Number: ");
        int accNo = sc.nextInt();
        System.out.println("Enter PIN: ");
        int enteredPin = sc.nextInt();

        for (BankAccount acc : accounts) {
            if (acc.getAccountNumber() == accNo && acc.getPin() == enteredPin) {
                return acc;
            }
        }
        return null;
    }

    // Show ATM menu
    public void startATM() {
        BankAccount account = login();

        if (account == null) {
            System.out.println("Invalid Account Number or PIN. Access Denied.");
            return;
        }

        while (true) {
            System.out.println("\n----- ATM Menu -----");
            System.out.println("1. Check Balance");
            System.out.println("2. Deposit");
            System.out.println("3. Withdraw");
            System.out.println("4. Exit");
            System.out.println("Choose an option: ");

            int choice = sc.nextInt();
            switch (choice) {
                case 1:
                    account.displayBalance();
                    break;
                case 2:
                    System.out.println("Enter deposit amount: ");
                    int dep = sc.nextInt();
                    account.deposit(dep);
                    break;
                case 3:
                    System.out.println("Enter withdraw amount: ");
                    int wit = sc.nextInt();
                    account.withdraw(wit);
                    break;
                case 4:
                    System.out.println("Thank you for using ATM!");
                    return;
                default:
                    System.out.println("Invalid choice! Try again.");
            }
        }
    }
}
